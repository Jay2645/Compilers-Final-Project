prog -> program id ; var dec_list begin stat-list end.
id -> letter id-after
id-after -> letter id-after
id-after -> digit id-after
id-after -> LAMBDA
dec-list -> dec : type ;
dec -> id , dec-prime
dec-prime -> id , dec-prime
dec-prime -> LAMBDA
type -> integer
stat-list-> stat
stat-list-> stat stat-list
stat -> write
stat -> assign
write -> print( id );
assign -> id = expr ;
expr -> term expr-prime
expr-prime -> + term expr-prime
expr-prime -> - term expr-prime
expr-prime -> LAMBDA
term -> factor term-prime
term-prime -> * term term-prime
term-prime -> / term term-prime
term-prime -> LAMBDA
factor -> id
factor -> number
factor -> ( expr )
number -> sign digit number-after
number-after -> digit
number-after -> LAMBDA
sign -> +
sign -> -
sign -> LAMBDA
digit -> 0
digit -> 1
digit -> 2
digit -> 3
digit -> 4
digit -> 5
digit -> 6
digit -> 7
digit -> 8
digit -> 9
letter -> p
letter -> q
letter -> r
